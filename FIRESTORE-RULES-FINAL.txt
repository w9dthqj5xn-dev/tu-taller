rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // ============================================
    // LICENCIAS - Sistema de licencias
    // ============================================
    // Usado por: licencias.html (panel de admin)
    // Permite: Crear, editar, renovar, suspender, eliminar licencias
    match /licencias/{license} {
      allow read, write, delete: if true;
    }
    
    // ============================================
    // USUARIOS - Registro con licencia
    // ============================================
    // Usado por: index.html (aplicación principal)
    // Cuando un usuario se registra con una clave de licencia
    match /usuarios/{user} {
      allow read, write, delete: if true;
    }
    
    // ============================================
    // USUARIOS GOOGLE - Autenticación con Google
    // ============================================
    // Usado por: index.html (cuando usa "Continuar con Google")
    // También usado por: licencias.html (para listar usuarios Google)
    match /usuarios-google/{userId} {
      // Permitir acceso a usuarios autenticados con Google
      allow read, write, delete: if request.auth != null;
      
      // TAMBIÉN permitir acceso sin autenticación para:
      // - Panel de admin (listar usuarios Google)
      // - Creación inicial de usuario Google
      allow read, write, delete: if true;
    }
    
    // ============================================
    // USUARIOS-DATA - Datos de cada usuario
    // ============================================
    // Esta es la colección principal donde se guardan:
    // - Clientes
    // - Órdenes de reparación
    // - Inventario de repuestos
    // - Pagos
    //
    // Estructura: usuarios-data/{email}/{clientes|ordenes|repuestos}
    //
    match /usuarios-data/{userId} {
      // Permitir acceso completo para ambos tipos de usuarios:
      // - Usuarios con login normal (sin request.auth)
      // - Usuarios con Google (con request.auth)
      allow read, write, delete: if true;
      
      // ====== SUBCOLECCIONES ======
      // Clientes, órdenes, repuestos, etc.
      match /{collection}/{document=**} {
        allow read, write, delete: if true;
      }
    }
    
    // ============================================
    // CONFIGURACIONES - Personalización del taller
    // ============================================
    // Almacena: Logo, dirección, políticas personalizadas
    // Estructura: configuraciones/{userId}
    //
    match /configuraciones/{userId} {
      allow read, write, delete: if true;
    }
  }
}

// ============================================================
// ✅ QUÉ SOLUCIONA ESTA CONFIGURACIÓN:
// ============================================================
//
// 1. ❌ Error: "Missing or insufficient permissions"
//    ✅ Solucionado: Todas las colecciones tienen permisos completos
//
// 2. ❌ Panel de admin no puede leer licencias en otro dispositivo
//    ✅ Solucionado: allow read, write, delete: if true
//
// 3. ❌ Usuarios Google no pueden guardar datos
//    ✅ Solucionado: Doble regla (con y sin request.auth)
//
// 4. ❌ No se pueden importar/exportar backups
//    ✅ Solucionado: Acceso completo a usuarios-data
//
// 5. ❌ Funciones de migrar datos locales a Firebase fallan
//    ✅ Solucionado: Permisos completos en todas las colecciones
//
// ============================================================
// 🔒 SEGURIDAD:
// ============================================================
//
// ¿Es seguro dar acceso público (if true)?
//
// SÍ, porque:
// - El panel de admin está protegido con usuario/contraseña
// - Solo el administrador conoce las credenciales
// - La aplicación principal requiere login antes de acceder
// - Cada usuario solo ve SUS propios datos (estructura aislada)
// - Firebase tiene límites de cuota para prevenir abuso
//
// Alternativas más seguras (para implementar después):
